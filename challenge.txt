/*      13. Which actor has appeared in the most films? */

FROM actor WHERE actor_id=(
	SELECT COUNT(film_id) AS number_of_films 
	FROM film_actor GROUP BY actor_id ORDER BY number_of_films DESC LIMIT 1);
